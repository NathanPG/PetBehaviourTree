tree("Root")
	sequence
		CheckTimePass
		fallback
			while
				IsDay
				tree("DayBehavior")
			while
				IsNight
				tree("NightBehavior")


tree("DayBehavior")
	fallback
		while
			OwnerIsOut
			tree("DayOut")
		while
			IsSleeping
			tree("DaySleep")
		tree("DayNormal")

tree("DayOut")
	fallback
		while
			IsHungry
			EatFood
		race
			while
				IsLonely
				Sleep
			while 
				WantToPlay
				Sleep
			while
				IsTired
				Sleep

tree("DaySleep")
	fallback
		while
			IsHungry
			while
				not 
					IsTired
				HWakeup
		Snore
			

tree("DayNormal")
	fallback
		while
			IsHungry
			EatFood
		while
			IsLonely
			LookForPet
		while 
			WantToPlay
			LookOutside
		while
			IsTired
			Sleep


tree("NightBehavior")
	fallback
		while
			OwnerIsOut
			tree("NightOut")
		while
			IsSleeping
			tree("NightSleep")
		tree("NightNormal")

tree("NightOut")
	fallback
		while
			IsHungry
			EatFood
		race
			while
				IsLonely
				Sleep
			while 
				WantToPlay
				Sleep
			while
				IsTired
				Sleep

tree("NightSleep")
	fallback
		fallback
		while
			IsLonely
			while
				not 
					IsTired
				LWakeup
		Snore
		

tree("NightNormal")
	fallback
		while
			IsTired
			Sleep
		while
			IsHungry
			EatFood
		while
			IsLonely
			LookForPet
		while 
			WantToPlay
			LookOutside
